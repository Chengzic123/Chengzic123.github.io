C++STL-vector

2022-10-17 C++STL 874 字
STL

头文件

#include<vector>

定义与初始化

一维数组

vector<int> a(n,m);
vector<int> a={1,2,3,4,5};
vector<int> b=a;

**第一行为定义了一个名为a，长度为n，初始值全为m的vector数组；其中<>之中可以填double，int（包括结构体）等用来定义变量的东西；第二行为赋值初始化;第三行为把a的元素全部复制给a；

二维数组

vector<int> a[n];

**此为定义了一个半动态二维数组，即列数可变，行数不能变；

vector<vector<int>> a

这样就定义了一个全动态二维数组，对其初始化需采用如下方式（我叫它复合赋值）；

vector<vector<int>> a(5,vector<int>(10,0));
vector a(5,vector<int>(10));//太新了尽量别用

访问与输出

下标循环遍历

**可以像普通数组一样循环遍历

vector<int> a(n);//或者可以使用.size()取得数组长度
for(int i=0;i<n;i++)
{
	cout<<a[i];
}

也可以使用比较高级的C++的智能指针

vector<int> a(n);
for(int i:a)
{
	cout<<i<<' ';
}
//或者
for(auto i:a)
{
	cout<<i<<' ';
}
//auto是可以自动判断变量类型的

使用迭代器访问

*迭代器就是C++里的指针，使用i储存数组a的地址，使用 i输出元素

vector<int> a;
vector<int>::iterator i=a.begin(); 
cout<<*i<<' ';
for(i=a.begin();i!=a.end();i++)
{
cout<<*i<<' ';
}
//或者
auto i=a.begin();
for(auto i:a)
{
cout<<i<<' ';
}

相关函数

c.front() ：

==返回第一个数据==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	cout<<c.front();
	return 0;
 } 
 //输出为 1

c.pop_back()：

==删除最后一个数据==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	c.pop_back();
	for(auto i:c)
	cout<<i<<' ';
	return 0;
 } 
 //输出为 1 2 3 4

c.push_back(element)：

==在最后添加一个元素==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	c.push_back(6);
	for(auto i:c)
	cout<<i<<' ';
	return 0;
 } 
 //输出为 1 2 3 4 5 6

c.clear()；c.empty()：

==清空元素；判断是否为空==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	cout<<c.empty()<<' ';
	c.clear();
	cout<<c.empty();
	return 0;
 } 
 //输出为 1 0

c.resize(n,v)：

==改变数组大小为n，n个空间赋值为v(若已赋值则不变成v)==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	c.resize(6,2);
	for(auto i:c)
	cout<<i<<' ';
	return 0;
 } 
 //输出为 1 2 3 4 5 2

c.insert(it,x)：

==向迭代器it中插入x(原本在这个位置上的元素依次往后挪)==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	c.insert(c.begin()+2,2);
	for(auto i:c)
	cout<<i<<' ';
	return 0;
 } 
 //输出为 1 2 2 3 4 5

c.erase(first,last)：

==删除从first到last的所有元素==

#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<int> c={1,2,3,4,5};
	c.erase(c.begin()+2,c.end());
	for(auto i:c)
	cout<<i<<' ';
	return 0;
 } 
 //输出为 1 2
